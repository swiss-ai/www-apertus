{{/* button shortcode Theme-aware button component that uses the Lotus Docs / Bootstrap styling. Usage (link button): {{< button href="https://publicai.co" icon="rocket_launch" style="primary" # e.g. primary, secondary, outline-primary, link size="lg" # sm | md | lg external="true" # "true" / "1" => target="_blank" rel="noopener noreferrer" id="hero-demo-btn" class="me-2 mt-2" >}} Try the Demo {{< /button >}} Usage (button element, no href): {{< button style="outline-primary" size="sm" type="button" >}} Learn More {{< /button >}} Notes: - Uses Bootstrap-style classes: e.g. "btn btn-lg btn-primary", "btn btn-lg btn-outline-primary". - `style` is appended as-is after "btn-". For example: style="primary" => class="btn btn-lg btn-primary" style="outline-primary" => class="btn btn-lg btn-outline-primary" - The inner content becomes the button label and can contain inline HTML. */}} {{- $href := .Get "href" | default "" -}} {{- $style := .Get "style" | default "primary" -}} {{- $size := .Get "size" | default "lg" -}} {{- $icon := .Get "icon" | default "" -}} {{- $iconPos := .Get "icon_position" | default "left" -}} {{- $id := .Get "id" | default "" -}} {{- $class := .Get "class" | default "" -}} {{- $externalArg := .Get "external" | default "" -}} {{- $type := .Get "type" | default "button" -}} {{- $blockArg := .Get "block" | default "" -}} {{- $isExternal := or (eq (lower $externalArg) "true") (eq $externalArg "1") -}} {{- $isBlock := or (eq (lower $blockArg) "true") (eq $blockArg "1") -}} {{- $btnClasses := slice "btn" -}} {{- if or (eq $size "sm") (eq $size "lg") -}} {{- $btnClasses = $btnClasses | append (printf "btn-%s" $size) -}} {{- end -}} {{- if $style }} {{- $btnClasses = $btnClasses | append (printf "btn-%s" $style) -}} {{- end -}} {{- if $isBlock }} {{- $btnClasses = $btnClasses | append "w-100" -}} {{- end -}} {{- if $class }} {{- $btnClasses = $btnClasses | append $class -}} {{- end -}} {{- $label := (trim .Inner " \n\r\t") -}} {{- $renderIconLeft := and $icon (ne (lower $iconPos) "right") -}} {{- $renderIconRight := and $icon (eq (lower $iconPos) "right") -}} {{- if $href }} {{- if $renderIconLeft }} {{ $icon }} {{- end }} {{ $label | safeHTML }} {{- if $renderIconRight }} {{ $icon }} {{- end }} {{- else }} {{- end }}
